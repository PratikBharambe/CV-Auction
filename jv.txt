class Solution {
    public int[] finalPrices(int[] p) {
        int n=p.length;
        for(int i=0; i < n; i++){
            int index = -1;
            for(int j=0; j < n; j++){
                if(p[i] > p[j]){
                    index = j;
                    break;
                }
            }
            p[i] -= p[index];
        }
        return p;
    }
}
