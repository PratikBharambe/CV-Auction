class Solution {
    public String compressedString(String word) {
        return function(word);
    }
    public String function(String w){
        StringBuilder word = new StringBuilder(w);
        StringBuilder str = new StringBuilder();
        
        while(word.length() != 0){
            int count = 1;
            char current = word.charAt(0);
            word.deleteCharAt(0);

            if( word.length() != 0 && current == word.charAt(0)){
                while(current == word.charAt(0)){
                    count++;
                    word.deleteCharAt(0);
                }
            }
            if(count <= 9){
                str.append(count);
                str.append(current);
            }else{
                while(count > 0){
                    int rem = count % 9;
                    count -= 9;
                    str.append(rem);
                    str.append(current);
                }
            }
        }
        return str.toString();
    }
}
